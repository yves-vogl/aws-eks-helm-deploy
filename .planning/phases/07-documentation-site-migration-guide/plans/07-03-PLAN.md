   ```python
   #!/usr/bin/env python3
   """Generate docs/reference/variables.md from Settings via settings-doc.

   Adds the D5 autogenerated-banner comment that settings-doc itself does not emit.

   import subprocess

   BANNER = (

       "— DO NOT EDIT BY HAND. Regenerate via "

   def main() -> int:

       body = subprocess.check_output(

               "--class", "aws_eks_helm_deploy.settings.Settings",
               "--output-format", "markdown",
           ],
           text=True,
       )
       TARGET.parent.mkdir(parents=True, exist_ok=True)
       TARGET.write_text(BANNER + body)
       return 0


   if __name__ == "__main__":
       sys.exit(main())
   ```
