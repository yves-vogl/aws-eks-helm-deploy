  push:

permissions:
  contents: write          # release-please pushes the release branch
  pull-requests: write     # release-please opens/updates the Release PR

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: false   # don't cancel a release-please run mid-PR-update

jobs:
  release-please:
    name: release-please
    runs-on: ubuntu-24.04
    steps:
      - name: Checkout
        uses: actions/checkout@11bd71901bbe5b1630ceea73d27597364c9af683 # v4.2.2

      - name: Run release-please
        uses: googleapis/release-please-action@45996ed1f6d02564a971a2fa1b5860e934307cf7 # v5.0.0

          # release-please reads the JSON config + manifest from the repo root

          token: ${{ secrets.GITHUB_TOKEN }}
